-- 1. Create divisions table
CREATE TABLE IF NOT EXISTS queueing_divisions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,
    display_name VARCHAR(100) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 2. Alter existing columns in queueing_queue_items and queueing_users to support longer division names
ALTER TABLE queueing_queue_items MODIFY COLUMN division VARCHAR(50);
ALTER TABLE queueing_users MODIFY COLUMN division VARCHAR(50);
